实验一

CREATE DATABASE student
ON PRIMARY (
   NAME = student_data,
   FILENAME = 'E:\sql_data\student_data.mdf',
   SIZE = 10MB,
   MAXSIZE = 50MB,
   FILEGROWTH = 5%
)
LOG ON (
   NAME = student_log,
   FILENAME = 'E:\sql_data\student_log.ldf',
   SIZE = 2MB,
   MAXSIZE = 5MB,
   FILEGROWTH = 1MB
);
CREATE TABLE student(
    sno CHAR(5) PRIMARY KEY CONSTRAINT student_id CHECK (sno like '[1-9][0-9][0-9][0-9][0-9]'),
    sname VARCHAR(10) NOT NULL,
    sex CHAR(2) CONSTRAINT student_sex CHECK (sex IN ('男','女')),
    native char(20),
    birthday smalldatetime,
    classno char(6),
    entime smalldatetime constraint student_time DEFAULT GETDATE(),
    home varchar(40),
    tel char(11) constraint student_tel check (tel like '1[385][0-9][0-9][0-9][0-9][0-9][0-9][0-9][0-9][0-9]'),
    age char(10) not null
);
CREATE TABLE course(
    cno CHAR(3) PRIMARY KEY,
    cname VARCHAR(20) NOT NULL,
    lecture int CONSTRAINT course_lecture CHECK (lecture between 12 and 90),
    semester int CONSTRAINT course_semester CHECK (semester LIKE '[1-8]'),
    credit int,
    type varchar(10) CONSTRAINT course_type CHECK (type IN ('公共课','专业基础课','专业课','选修课')),
);
CREATE TABLE sc(
    sno CHAR(5),
    cno CHAR(3) NOT NULL,
    grade int CONSTRAINT sc_grade CHECK (grade between 0 and 100),
    constraint sc_majorkey primary key (sno,cno),
    constraint sc_sno_majorkey foreign key (sno) references student(sno),
    constraint sc_cno_majorkey foreign key (cno) references course(cno),
);
CREATE TABLE depart(
    dno CHAR(6) primary key,
    dname VARCHAR(10) NOT NULL,
    tel CHAR(6),
);
CREATE TABLE class(
    classno CHAR(6) primary key,
    classname VARCHAR(10) NOT NULL,
    dno CHAR(6),
    constraint class_dno_majorkey foreign key (dno) references depart(dno)
);
ALTER TABLE course ADD CONSTRAINT course_experiment_majorkey CHECK (experiment between 6 and 24);

-- 14.
-- 14.1 男性或大于19的姓名,课程号,成绩
SELECT sname, cno, grade
FROM sc,student
WHERE (sex = '男'
    OR age > 19)
  AND sc.sno = student.sno
-- 14.2 两个表联合
select sname, cname, grade
from course,
     (SELECT sname, cno, grade
      FROM sc,
           student
      WHERE (sex = '男'
          OR age > 19)
        AND sc.sno = student.sno) AS sons
where course.cno = sons.cno;

-- 15
-- 15.1 查询出的每个学生的学号,平均分和总分
select sno, AVG(grade) AS 平均分, SUM(grade) AS 总分
from sc
group by sno;
-- 15.1 查询出的每个学生的名字,平均分和总分
select sname, sons.平均分, sons.总分
from student,
     (select sno, AVG(grade) AS 平均分, SUM(grade) AS 总分
      from sc
      group by sno) AS sons
where student.sno = sons.sno;

-- 16
-- 16.1先查询每个学生个选修了几门课
select sno, count(cno) AS 选修的课程数
from sc
group by sno;
-- 16.2查询一共有多少门课
select count(cno)
from course;
-- 16.3 将表联合
select DISTINCT  sc.sno
from sc,
     (select sno, count(cno) AS 选修的课程数 from sc group by sno) AS sons
where sc.sno = sons.sno
  AND sons.选修的课程数 = (select count(cno) from course);
-- 16.4继续将表联合
select DISTINCT  sname
from student,
     (select sc.sno
      from sc,
           (select sno, count(cno) AS 选修的课程数 from sc group by sno) AS sons
      where sc.sno = sons.sno
        AND sons.选修的课程数 = (select count(cno) from course)) AS sons
where student.sno = sons.sno;

-- 1
CREATE VIEW V_jsj AS
SELECT student.*, class.classname
FROM student
         JOIN class ON student.sno = class.classno;
-- 2
CREATE VIEW S_C_GRADE AS
SELECT s.sno, s.sname, c.cname, sc.grade
FROM student s
         JOIN sc sc ON s.sno = sc.sno
         JOIN course c ON sc.cno = c.cno;
-- 3
CREATE VIEW COMPUTE_AVG_GRADE AS
SELECT s.sno, c.cno, AVG(sc.grade) AS avg_grade
FROM student s
         JOIN sc sc ON s.sno = sc.sno
         JOIN course c ON sc.cno = c.cno
WHERE s.classno = '1'
GROUP BY s.sno, c.cno;
-- 4
CREATE VIEW V_AVG_S_G AS
SELECT s.sno, COUNT(sc.cno) AS course_num, AVG(sc.grade) AS avg_grade
FROM student s
         JOIN sc sc ON s.sno = sc.sno
GROUP BY s.sno;
-- 5.
SELECT COUNT(cno) AS course_num, AVG(grade) AS avg_grade
FROM sc
WHERE sno = '95001';
-- 6.
SELECT s.*, c.classname
FROM student s
         JOIN class c ON s.classno = c.classno
WHERE c.classno = 'A001';

-- 8
create table stuback
(
    sno      char(5)       not null
        primary key
        constraint student_ids
            check ([sno] like '[1-9][0-9][0-9][0-9][0-9]'),
    sname    varchar(10)   not null,
    sex      char(2)
        constraint student_sexs
            check ([sex] = '女' OR [sex] = '男'),
    native   char(20),
    birthday smalldatetime not null,
    classno  char(6),
    entime   smalldatetime
        constraint student_times default getdate(),
    home     varchar(40),
    tel      char(11)
        constraint student_tels
            check ([tel] like '1[385][0-9][0-9][0-9][0-9][0-9][0-9][0-9][0-9][0-9]'),
    age      int
);
CREATE TRIGGER tr_insert ON student
AFTER INSERT
AS
BEGIN
    INSERT INTO stuback(sno, sname, sex,native,birthday,classno,entime,home,tel,age)
    select sno, sname, sex,native,birthday,classno,entime,home,tel,age
    FROM inserted;
END
-- 9
CREATE PROCEDURE P_CX
    @sno VARCHAR(10)
AS
BEGIN
    IF @sno IS NULL
    BEGIN
        PRINT '请输入学号'
    END
    ELSE IF NOT EXISTS (SELECT * FROM S_C_GRADE WHERE sno = @sno)
    BEGIN
        PRINT '学号不存在'
    END
    ELSE
    BEGIN
        SELECT s.sname, c.cname, sc.grade
        FROM sc sc
        JOIN student s ON sc.sno = s.sno
        JOIN course c ON sc.cno = c.cno
        WHERE sc.sno = @sno
    END
END
-- 10
CREATE PROCEDURE P_CJTJ
    @sno VARCHAR(10)
AS
BEGIN
    IF @sno IS NULL
    BEGIN
        PRINT '请输入学号'
    END
    ELSE IF NOT EXISTS (SELECT * FROM S_C_GRADE WHERE sno = @sno)
    BEGIN
        PRINT '学号不存在'
    END
    ELSE
    BEGIN
        SELECT s.sname,COUNT(cno) ,AVG(sc.grade),SUM(sc.grade),MAX(sc.grade),MIN(sc.grade)
        FROM sc sc
        JOIN student s ON sc.sno = s.sno
        JOIN course c ON sc.cno = c.cno
        WHERE sc.sno = @sno
        group by s.sname
    END
END
-- 11
CREATE TRIGGER tr_check_course_limit
ON sc
AFTER INSERT
AS
BEGIN
    -- 检查学生的课程数量是否超过了3门
    DECLARE @n INT;
    SELECT sno sno from inserted;
    IF EXISTS (
        SELECT COUNT(cno) FROM student
        where sc.sno=@n
    )
    BEGIN
        RAISERROR('每个学生最多只能选3门课程', 16, 1);
        ROLLBACK;
        RETURN;
    END;
END;
