平衡二叉树

案例-二叉树可能存在的问题

给你一个数列{1,2,3,4,5,6}，要求创建一颗二叉排序树(BST), 并分析问题所在.

·左边 BST 存在的问题分析:
1.左子树全部为空，从形式上看，更像一个单链表.
2。插入速度没有影响
3.查询速度明显降低(因为需要依次比较), 不能发挥 BST的优势，因为每次还需要比较左子树，其查询速度比单链表还慢
4.解决方案-平衡二叉树

基本介绍

1.平衡二叉树也叫做二叉搜索树又被称为AVL树，可以保证查询效率较高
2.具有一下特点,它是一棵空树或它的左右两个子树的高度查的绝对值不超过1,并且左右两个子树都是一颗平衡二叉树

左旋转

已该图的树为例进行左旋转

1.创建一个新节点newNode【以该节点的值为新节点的值】
2.新节点的左子节点为该节点的左子节点 4.left=3
3.新节点的右子结点为该节点的右子节点的左子节点 4.right=5
4.该节点的值为该节点的右子节点的值 此时该节点的值为 6
5.该节点的左子节点为该节点的左子节点的左子节点 6.right=right
6.该节点的右子节点为新的结点 6.right=4

左旋转后的树

右旋转

已该图的树为例进行右旋转

1.创建一个新节点newNode【以该节点的值为新节点的值】
2.新节点的右子节点为该节点的右子节点 10.right=12
3.新节点的左子节点为该节点的左子节点的右子节点 10.left=9
4.该节点的值为该节点的左子结点的值10->8;
5.该节点的左子节点为该节点的左子节点的左子节点 10.left=7；
6.该节点的左子节点为新的结点 8.left=10

什么时候进行旋转操作

每添加一个节点时,如果左子结点的高度-右子节点的高度>1,向右旋转

每添加一个节点时,如果右子结点的高度-左子节点的高度>1,向左旋转

代码实现

package 数据结构与算法二.二叉平衡树;

/**
 * @Author 秦家大胖

 */
public class no1 {
    public static void main(String[] args) {
        int[] arr = {4, 3, 6, 5, 7, 8};
        //创建一个AVL树
        AVLTree avlTree = new AVLTree();
        //添加结点
        for (int i = 0; i < arr.length; i++) {
            avlTree.add(new Node(arr[i]));
        }
        //遍历
        System.out.println("中序遍历");
        avlTree.infixOrder();
    }
}

//创建AVLTree
class AVLTree {
    private Node root;

    //编写方法，删除节点
    public void delNode(int val) {
        if (root == null) {
            return;
        } else {
            //1.先得到要删除的节点
            Node target = search(val);
            if (target == null) {
                return;
            }
            //如果我们发现当前的这颗二叉树只有一个节点时
            if (root.left == null && root.right == null) {
                root = null;
                return;
            }
            //2.找到要删除节点的父节点，因为树没有自我删除的能力，只要链表有自我删除的能力
            Node parent = searchNode(val);
            //要删除节点的三种情况
            //（1）如果要删除的节点是叶子节点，无子节点
            if (target.left == null && target.right == null) {
                //判断要删除的节点是它父节点的左子节点还是右子节点
                if (parent.left != null && parent.left.value == val) {
                    parent.left = null;
                } else if (parent.right != null && parent.right.value == val) {
                    parent.right = null;
                }
                //(2)如果要删除的节点有两颗子树
            } else if (parent.left != null && parent.right != null) {
                //找出当前节点右子树的最小值，并删除
                int minVal = delLeftTreeMin(target.right);
                target.value = minVal;
            } else {//要删除的节点只要一颗子树
                //如果要删除的节点有左子树
                if (target.left != null) {
                    //判断要删除的节点是它父节点的左子节点还是右子节点
                    if (parent != null) {
                        if (parent.left.value == val) {//如果targetNode是parent的左子节点
                            parent.left = target.left;
                        } else {
                            parent.right = target.left;
                        }
                    } else {
                        root = target.left;
                    }
                    //如果要删除的节点有左子树
                } else {
                    if (parent != null) {
                        if (parent.left.value == val) {//如果targetNode是parent的左子节点
                            parent.left = target.right;
                        } else {
                            parent.right = target.right;
                        }
                    } else {
                        root = target.right;
                    }
                }
            }
        }
    }

    //编写方法
    //1.返回的以node为根节点的二叉排序树的最小节点的值
    //2.删除node为根节点的二叉排序树的最小节点
    public int delLeftTreeMin(Node node) {
        Node temp = node;
        while (temp.left != null) {
            temp = temp.left;
        }
        //这是temp的值就是最小的值
        //删除最小的值
        delNode(temp.value);
        return temp.value;
    }

    //查找要删除节点的父节点
    public Node searchNode(int val) {
        if (root == null) {
            return null;
        } else {
            return root.searchParent(val);
        }
    }

    //查找要删除的节点
    public Node search(int val) {
        if (root == null) {
            return null;
        } else {
            return root.search(val);
        }
    }

    //向平衡二叉树中添加节点
    public void add(Node node) {
        if (root == null) {
            root = node;
        } else {
            root.add(node);
        }
    }

    // 中序遍历
    public void infixOrder() {
        if (root != null) {
            root.infixOrder();
        } else {
            System.out.println("数为空");
        }
    }
}

class Node {
    int value;
    Node left;
    Node right;

    public Node(int value) {
        this.value = value;
    }

    //返回左子树的高度
    public int leftHeight() {
        if (left == null) {
            return 0;
        }
        return left.height();
    }

    //返回右子树的高度
    public int rightHeight() {
        if (right == null) {
            return 0;
        }
        return right.height();
    }

    //返回以该节点为根节点的树的高度
    public int height() {
        return Math.max(left == null ? 0 : left.height(), right == null ? 0 : right.height())+1;
    }
    //左旋转的方法
    private void leftRotate(){
        //创建新的结点，以当前根节点的值
        Node newnode = new Node(value);
        //把新的结点的左子树设置为当前结点的左子树
        newnode.left=this.right.left;
        //把新的结点的右子树设置为当前结点的
        newnode.right=this.right.left;
        //当前结点的值为当前结点的右子节点的值
        this.value=this.right.value;
        this.left=this.left.left;
    }
    //右旋转
    private void rightRotate(){
        //创建新的结点,以当前结点的值
        Node newNode = new Node(value);
        //把新结点的左子树设置当前结点的左子树
        newNode.right=right;
        //把新结点的右子树设置成当前节点的右子树的左子树
        newNode.right=right.left;
        //把当前节点的值替换成右子节点的值
        value=right.value;
        //当前节点的右子结点设置成当前结点右子结点的右子结点
        right=right.right;
        //当前结点的右子节点指向新的结点
        left=newNode;
    }
    //查找要删除的结点
    /**
     * @param val 希望删除的节点的值
     * @return 如果找到返回该节点，否则返沪null
     */
    public Node search(int val) {
        if (val == this.value) {//找到就是该节点
            return this;
        } else if (val < this.value) {//如果查找的值小于当前的结点，向左子树递归查找
            if (this.left == null) {
                return null;
            }
            return this.left.search(val);
        } else {//如果查找的值不小于当前的结点，向右子树递归查找
            if (this.right == null) {
                return null;
            }
            return this.right.search(val);
        }
    }
    //查找要删除结点的父节点

    /**
     * @param val 要找到的节点的值
     * @return 返回的是要删除的结点的父节点，如果没有就返回null
     */
    public Node searchParent(int val) {
        //如果当前节点就是要删除结点的父节点，就返回
        if ((this.left != null && this.left.value == val) || (this.right != null && this.right.value == val)) {
            return this;
        } else {
            //如果查找的值小于当前节点的值，并且当前节点的左子节点不为空
            if (val < this.value && this.left != null) {
                return this.left.searchParent(val);
            } else if (val >= this.value && this.right != null) {
                return this.right.searchParent(val);//相右子树递归查找
            } else {
                return null;
            }
        }
    }

    // 添加节点的方法
    // 递归的形式添加节点，注意需要满足二叉排序树的要求
    public void add(Node node) {
        if (node == null) {
            return;
        }
        // 判断传入的结点的值，和当前子树的根节点的关系
        if (node.value < this.value) {
            // 如果当前结点左子树节点为null
            if (this.left == null) {
                this.left = node;
            } else {
                // 递归的向左子树添加
                this.left.add(node);
            }
        } else {// 添加的结点的值大于当前结点的值
            if (this.right == null) {
                this.right = node;
            } else {// 递归的向右子树添加
                this.right.add(node);
            }
        }
        //当添加完一个节点后,如果(右子树的高度-左子树的高度)>1,左旋转
        if (rightHeight()-leftHeight()>1) {
            //如果它的右子树的左子树高度大于右子树的左子树高度，先将右子树右旋转
            if (right!=null&&right.leftHeight()>right.rightHeight()){
                right.rightRotate();
                leftRotate();//左旋转
            }else{
                leftRotate();
            }
            return;//必须要
        }
        if (leftHeight()-rightHeight()>1){
            //如果它的左子树的右子树高度大于左子树的左子树高度，先左旋转，在右旋转
            if(left!=null&&left.rightHeight()>left.leftHeight()){
                left.leftRotate();//左旋转
                rightRotate();//右旋转
            }else{
                rightRotate();//右旋转
            }
        }
    }

    // 中序遍历
    public void infixOrder() {
        if (this.left != null) {
            this.left.infixOrder();
        }
        System.out.println(this);
        if (this.right != null) {
            this.right.infixOrder();
        }
    }

    @Override
    public String toString() {
        return "Node{" +
                "value=" + value +
                '}';
    }
}
