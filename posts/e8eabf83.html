-- 1.
select sno, sname
from student
where classno = 'A001';

-- 2.
select sno, cno
from sc
where cno is not null;

-- 3.
select sno, grade
from sc
where cno = '001'
order by grade DESC, sno ASC;

-- 4.
select sno, grade * 0.75 AS grade
from sc
where cno = '001'
  AND grade between 80 and 90;

-- 5.
select *
from student
where classno in ('A001', 'A002')
  AND sname like '张%';

-- 6.
select sno, cno
from sc
where grade is null;

-- 7.
SELECT course.cno, COUNT(sc.sno) AS 选课人数
FROM course
LEFT JOIN sc ON course.cno = sc.cno
GROUP BY course.cno;

-- 8.
select classno, count(sno)
from student
group by classno;

-- 9.
select student.sname, sc.cno, sc.grade
from student,sc
where student.sno = sc.sno;

-- 10
SELECT sno
FROM sc
WHERE cno = '001'
  AND sno IN (SELECT sno FROM sc
              WHERE cno = '002');

-- 11.
SELECT student.sname, AVG(grade) AS avg_grade
FROM sc INNER JOIN student ON sc.sno = student.sno
GROUP BY student.sno, student.sname
HAVING AVG(sc.grade) > 85;

-- 12.
select sname from student
inner join sc on student.sno = sc.sno
where cno = '001';

-- 13.
select cname from course
inner join sc on sc.cno = course.cno
where sno = '95002';

-- 14.
-- 14.1 男性或大于19的姓名,课程号,成绩
SELECT sname, cno, grade
FROM sc,student
WHERE (sex = '男'
    OR age > 19)
  AND sc.sno = student.sno
-- 14.2 两个表联合
select sname, cname, grade
from course,
     (SELECT sname, cno, grade
      FROM sc,
           student
      WHERE (sex = '男'
          OR age > 19)
        AND sc.sno = student.sno) AS sons
where course.cno = sons.cno;

-- 15
-- 15.1 查询出的每个学生的学号,平均分和总分
select sno, AVG(grade) AS 平均分, SUM(grade) AS 总分
from sc
group by sno;
-- 15.1 查询出的每个学生的名字,平均分和总分
select sname, sons.平均分, sons.总分
from student,
     (select sno, AVG(grade) AS 平均分, SUM(grade) AS 总分
      from sc
      group by sno) AS sons
where student.sno = sons.sno;

-- 16
-- 16.1先查询每个学生个选修了几门课
select sno, count(cno) AS 选修的课程数
from sc
group by sno;
-- 16.2查询一共有多少门课
select count(cno)
from course;
-- 16.3 将表联合
select DISTINCT  sc.sno
from sc,
     (select sno, count(cno) AS 选修的课程数 from sc group by sno) AS sons
where sc.sno = sons.sno
  AND sons.选修的课程数 = (select count(cno) from course);
-- 16.4继续将表联合
select DISTINCT  sname
from student,
     (select sc.sno
      from sc,
           (select sno, count(cno) AS 选修的课程数 from sc group by sno) AS sons
      where sc.sno = sons.sno
        AND sons.选修的课程数 = (select count(cno) from course)) AS sons
where student.sno = sons.sno;

-- 17.
update student set age=21 where sno='95002';

-- 18
-- 18.1先查询李丽在那个班
select classno
from student
where sname = '张三';
-- 18.2将表联合
select *
from student
where classno = (select classno from student where sname = '张三');

-- 19
-- 19.1 先找到1班的学生的学号
select sno
from student
where classno = 'A001';
-- 19.2 在将1班的学生成绩都加10分
update sc
set grade=grade + 10 from sc
join (select sno from student where classno = 'A001') AS sons on sc.sno = sons.sno;

-- 20
-- 20.1 先找到未选课的学生的学号
select sno from sc
where cno is null;
-- 20.2 将两个表联合
delete from student
from student
join (select sno from sc where cno is null) AS sons on student.sno = sons.sno;
