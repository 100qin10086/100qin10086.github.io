数据库的第二次实验：注意由于实验未给出任何表与字段，所以下面代码中的表和字段均为自己设计，所以在某几个题中会出现一些问题，由于数据和表均为自己设计，所以在参考时尽可能不要一致

如果有能力可以在我的实验结构上进行改进

CREATE TABLE student
(
    sno       VARCHAR(10) PRIMARY KEY, -- 学号，主键
    sname     VARCHAR(20)  NOT NULL,   -- 姓名，不能为空
    sex       CHAR(2)      NOT NULL,   -- 性别，不能为空
    born      DATE         NOT NULL,   -- 出生日期，不能为空
    telephone VARCHAR(20),             -- 电话号码，可为空
    address   VARCHAR(100) NOT NULL,   -- 地址，不能为空
    content   TEXT,                    -- 备注，可为空
    classno   VARCHAR(20)  NOT NULL    -- 班级编号，不能为空
);
CREATE TABLE course
(
    cno    VARCHAR(10) PRIMARY KEY, -- 课程编号，主键
    cname  VARCHAR(20) NOT NULL,    -- 课程名称，不能为空
    cpno   VARCHAR(10),             -- 先修课程编号，可为空
    credit INT         NOT NULL     -- 学分，不能为空
);
CREATE TABLE sc
(
    sno   VARCHAR(10) NOT NULL,                 -- 学号，不能为空
    cno   VARCHAR(10) NOT NULL,                 -- 课程编号，不能为空
    grade INT,                                  -- 成绩，
    PRIMARY KEY (sno, cno),                     -- 学号和课程编号组成主键
    FOREIGN KEY (sno) REFERENCES student (sno), -- 学号为外键，参考 student 表
    FOREIGN KEY (cno) REFERENCES course (cno)   -- 课程编号为外键，参考 course 表
);

-- 1.查询一班的学生学号和姓名
select sno, sname
from student
where classno = 'A001';
-- 2.查询选修了课程的学号和课程号
select sno, cno
from sc
where cno is not null;
-- 3.
select sno, grade
from sc
where cno = '002'
order by grade DESC, sno ASC;
-- 4.
select sno, grade * 0.75 AS grade
from sc
where cno = '002'
  AND grade between 80 and 90;
-- 5.
select * from student
where classno in ('A001', 'A002')
  AND sname like '张%';
-- 6.
select sno, cno
from sc
where grade is null;
-- 7.
select course.cno, count(sc.sno) AS 选课人数
from sc,course
where sc.cno = course.cno
group by course.cno;
-- 8.
select classno, count(sno)
from student
group by classno;
-- 9.
select student.sname, sc.cno, sc.grade
from student,
     sc
where student.sno = sc.sno;
-- 10.
SELECT sno
FROM sc
WHERE cno = '001'
  AND sno IN (SELECT sno FROM sc
              WHERE cno = '002');
-- 11.
SELECT student.sname, AVG(grade) AS avg_grade
FROM sc INNER JOIN student ON sc.sno = student.sno
GROUP BY student.sno, student.sname
HAVING AVG(sc.grade) > 85;
-- 12.
select sname from student
inner join sc on student.sno = sc.sno
where cno = '001';
-- 13.
select cname from course
inner join sc on sc.cno = course.cno
where sno = '002';
-- 14
-- 14.1 男性或大于19的姓名,课程号,成绩
SELECT sname, cno, grade
FROM sc,student
WHERE (sex = '男'
    OR (DATEDIFF(YEAR, born, GETDATE()) > 19))
  AND sc.sno = student.sno
-- 14.2 两个表联合
select sname, cname, grade
from course,
     (SELECT sname, cno, grade
      FROM sc,
           student
      WHERE (sex = '男'
          OR (DATEDIFF(YEAR, born, GETDATE()) > 19))
        AND sc.sno = student.sno) AS sons
where course.cno = sons.cno;
-- 15
-- 15.1 查询出的每个学生的学号,平均分和总分
select sno, AVG(grade) AS 平均分, SUM(grade) AS 总分
from sc
group by sno;
-- 15.1 查询出的每个学生的名字,平均分和总分
select sname, sons.平均分, sons.总分
from student,
     (select sno, AVG(grade) AS 平均分, SUM(grade) AS 总分
      from sc
      group by sno) AS sons
where student.sno = sons.sno;

-- 16
-- 16.1先查询每个学生个选修了几门课
select sno, count(cno) AS 选修的课程数
from sc
group by sno;
-- 16.2查询一个有多少门课
select count(cno)
from course;
-- 16.3 将表联合
select sc.sno
from sc,
     (select sno, count(cno) AS 选修的课程数 from sc group by sno) AS sons
where sc.sno = sons.sno
  AND sons.选修的课程数 = (select count(cno) from course);
-- 16.4继续将表联合
select sname
from student,
     (select sc.sno
      from sc,
           (select sno, count(cno) AS 选修的课程数 from sc group by sno) AS sons
      where sc.sno = sons.sno
        AND sons.选修的课程数 = (select count(cno) from course)) AS sons
where student.sno = sons.sno;

-- 19.1 先找到1班的学生的学号
select sno
from student
where classno = 'A001';
-- 19.2 在将1班的学生成绩都加10分
update sc
set grade=grade + 10 from sc
join (select sno from student where classno = 'A001') AS sons on sc.sno = sons.sno;

-- 20
-- 20.1 先找到未选课的学生的学号
select sno from sc
where cno is null;
-- 20.2 将两个表联合
delete from student
from student
join (select sno from sc where cno is null) AS sons on student.sno = sons.sno;
