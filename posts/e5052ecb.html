二叉排序树

需求

给你一个数列 (7, 3, 10, 12, 5, 1, 9)，要求能够高效的完成对数据的查询和添加

解决方案分析

· 使用数组
数组未排序， 优点：直接在数组尾添加，速度快。 缺点：查找速度慢.
数组排序，优点：可以使用二分查找，查找速度快，缺点：为了保证数组有序，在添加新数据时，找到插入位置后，后面的数据需整体移动，速度慢。
· 使用链式存储-链表
不管链表是否有序，查找速度都慢，添加数据速度比数组快，不需要数据整体移动。
· 使用二叉排序

二叉排序树介绍

1.二叉排序树：BST: (Binary Sort(Search) Tree), 对于二叉排序树的任何一个非叶子节点，要求左子节点的值比当前节点的值小，右子节点的值比当前节点的值大

二叉排序树的删除

二叉树的删除思路,删除的情况比较复杂，有三种情况需要考虑

1.删除叶子节点 (比如：2, 5, 9, 12)
2.删除只有一颗子树的节点 (比如：1)
3.删除有两颗子树的节点. (比如：7, 3，10 )

操作的思路分析

第一种情况：删除叶子节点

(1) 需求先去找到要删除的结点 targetNode
(2) 找到 targetNode 的 父结点 parent
(3) 确定 targetNode 是 parent 的左子结点 还是右子结点
(4) 根据前面的情况来对应删除
左子结点 parent.left = null
右子结点 parent.right = null

第二种情况：删除只有一棵子树的节点

(1) 需求先去找到要删除的结点 targetNode
(2) 找到 targetNode 的 父结点 parent
(3) 确定 targetNode 的子结点是左子结点还是右子结点
(4) targetNode 是 parent 的左子结点还是右子结点
(5) 如果 targetNode 有左子结点
5.1如果 targetNode 是 parent 的左子结点
parent.left = targetNode.left;
5.2 如果 targetNode 是 parent 的右子结点
parent.right = targetNode.left;
(6) 如果 targetNode 有右子结点
6.1 如果 targetNode 是 parent 的左子结点
parent.left = targetNode.right;
6.2 如果 targetNode 是 parent 的右子结点
parent.right = targetNode.right

第三种情况：删除有两棵子树的节点

(1) 需求先去找到要删除的结点 targetNode
(2) 找到 targetNode 的 父结点 parent
(3) 从 targetNode 的右子树找到最小的结点
(4) 用一个临时变量，将 最小结点的值保存 temp = 11
(5) 删除该最小结点
(6) targetNode.value = temp

//创建一个二叉树
class BinarySortTree {
    private Node root;

    public BinarySortTree(Node root) {
        this.root = root;
    }

    //编写方法，删除节点
    public void delNode(int val) {
        if (root == null) {
            return;
        } else {
            //1.先得到要删除的节点
            Node target = search(val);
            if (target == null) {
                return;
            }
            //如果我们发现当前的这颗二叉树只有一个节点时
            if (root.left == null && root.right == null) {
                root = null;
                return;
            }
            //2.找到要删除节点的父节点，因为树没有自我删除的能力，只要链表有自我删除的能力
            Node parent = searchNode(val);
            //要删除节点的三种情况
            //（1）如果要删除的节点是叶子节点，无子节点
            if (target.left == null && target.right == null) {
                //判断要删除的节点是它父节点的左子节点还是右子节点
                if (parent.left != null && parent.left.val == val) {
                    parent.left = null;
                } else if (parent.right != null && parent.right.val == val) {
                    parent.right = null;
                }
                //(2)如果要删除的节点有两颗子树
            } else if (parent.left != null && parent.right != null) {
                //找出当前节点右子树的最小值，并删除
                int minVal = delLeftTreeMin(target.right);
                target.val = minVal;
            } else {//要删除的节点只要一颗子树
                //如果要删除的节点有左子树
                if (target.left != null) {
                    //判断要删除的节点是它父节点的左子节点还是右子节点
                    if (parent != null) {
                        if (parent.left.val == val) {//如果targetNode是parent的左子节点
                            parent.left = target.left;
                        } else {
                            parent.right = target.left;
                        }
                    } else {
                        root = target.left;
                    }
                    //如果要删除的节点有左子树
                } else {
                    if (parent != null) {
                        if (parent.left.val == val) {//如果targetNode是parent的左子节点
                            parent.left = target.right;
                        } else {
                            parent.right = target.right;
                        }
                    } else {
                        root = target.right;
                    }
                }
            }
        }
    }

    //编写方法
    //1.返回的以node为根节点的二叉排序树的最小节点的值
    //2.删除node为根节点的二叉排序树的最小节点
    public int delLeftTreeMin(Node node) {
        Node temp = node;
        while (temp.left != null) {
            temp = temp.left;
        }
        //这是temp的值就是最小的值
        //删除最小的值
        delNode(temp.val);
        return temp.val;
    }

    //查找要删除节点的父节点
    public Node searchNode(int val) {
        if (root == null) {
            return null;
        } else {
            return root.searchParent(val);
        }
    }

    //查找要删除的节点
    public Node search(int val) {
        if (root == null) {
            return null;
        } else {
            return root.search(val);
        }
    }

    //向平衡二叉树中添加节点
    public void add(Node node) {
        if (root == null) {
            root = node;
        } else {
            root.add(node);
        }
    }

    // 中序遍历
    public void infixOrder() {
        if (root != null) {
            root.infixOrder();
        } else {
            System.out.println("数为空");
        }
    }
}

class Node {
    int val;
    Node left;
    Node right;

    public Node(int val) {
        this.val = val;
    }

    public int getVal() {
        return val;
    }

    public void setVal(int val) {
        this.val = val;
    }

    public Node getLeft() {
        return left;
    }

    public void setLeft(Node left) {
        this.left = left;
    }

    public Node getRight() {
        return right;
    }

    public void setRight(Node right) {
        this.right = right;
    }

    //查找要删除的节点的父节点
    public Node searchParent(int val) {
        //如果当前节点就是要查找节点的父节点就返回
        if ((this.left != null && this.left.val == val) || (this.right != null && this.right.val == val)) {
            return this;
        } else {
            //如果要查找的值小于当前节点值,并且当前节点的左子节点的值不为空
            if (val < this.val && this.left != null) {
                return this.left.searchParent(val);
            } else if (val >= this.val && this.right != null) {
                return this.right.searchParent(val);
            } else {
                return null;
            }
        }
    }

    //查找要删除的节点
    public Node search(int val) {
        if (val == this.val) {//如果要删除的节点为头节点
            return this;
        } else if (val < this.val) {
            if (this.left == null) {
                return null;
            }
            return this.left.search(val);
        } else {
            if (this.right == null) {
                return null;
            }
            return this.right.search(val);
        }
    }

    //添加节点的方法
    //递归的方式添加节点，注意要满足二叉排序树的要求
    public void add(Node node) {
        if (node == null) {
            return;
        }
        if (node.val < this.val) {
            if (this.left == null) {
                this.left = node;
            } else {
                this.left.add(node);
            }
        } else if (node.val > this.val) {
            if (this.right != null) {
                this.right = node;
            } else {
                this.right.add(node);
            }
        }
    }

    //中序遍历
    public void infixOrder() {
        if (this.left != null) {
            this.left.infixOrder();
        }
        System.out.print(this + "\t");
        if (this.right != null) {
            this.right.infixOrder();
        }
    }

    @Override
    public String toString() {
        return "Node{" +
                "val=" + val +
                '}';
    }
}
